	Header //generated by me because I cant find the header in the debug build :(
	{
0x00	uint32 m_ID 0x5346494C (SFIL)
0x04 	uint32 m_uiVersion
0x08	uint64 m_uiStamp
0x10 	uint32 m_uiOffset
0x14 	uint32 m_uiNumCollections //number of collections in this stream file
0x18	uint32 m_StreamFormat // 0 = Raw; 1 = Compressed; 2 = Compressed Arena; 3 = CompressedChunkArena
0x1c 	uint32 unknown //this seems to be the same as the number of collections in CMAP with a flipped endianess but sometimes it is just 0xFFFFFFFF?
	}

	Collection Asset
	{
0x00	uint64 m_ID
0x08 	uint32 m_uiSize
0x0C 	uint32 m_uiOffset //from start of this
0x10 	uint32 m_uiStride //offset to next asset
0x14 	uint32 unused
	}

		}
